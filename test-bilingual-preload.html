async function loadQuestions() {
    // 同時載入中文和英文版本的題目
    const [chineseResponse, englishResponse] = await Promise.all([
        fetch('data/questions.json'),
        fetch('data/questions_en.json')
    ]);
    
    const chineseQuestions = await chineseResponse.json();
    const englishQuestions = await englishResponse.json();
    
    // 創建雙語言題目集合
    allQuestionsBilingual = createBilingualQuestionSet(chineseQuestions, englishQuestions);
}

🧪 測試步驟

📋 雙語言預載入測試流程

  1. 預載入測試
    • 前往筆試測驗頁面
    • 開始測驗，檢查控制台日誌
    • 確認看到"Loaded bilingual questions"訊息
    • 確認雙語言題目集合已創建
  2. 選項順序一致性測試
    • 在中文模式下查看選項順序
    • 切換到英文模式
    • 確認選項順序完全相同
    • 切換回中文模式再次確認
  3. 即時切換測試
    • 在測驗中快速切換語言
    • 確認切換無延遲
    • 確認題目和選項立即更新
    • 確認計時器不受影響
  4. 狀態保持測試
    • 在答題過程中切換語言
    • 確認當前進度保持
    • 確認已作答題目狀態保持
    • 確認計時器狀態正確
